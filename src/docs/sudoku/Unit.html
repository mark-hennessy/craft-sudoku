Unit class

A grouping of cells such as a row, column or block.

class Unit {
 List<Cell> _cells = [];
 List<Cell> get cells => _cells.toList();

 List<int> get cellValues =>
     _cells.where((cell) => cell.hasValue)
     .map((cell) => cell.value).toList();

 String cssClass = '';

 void add(Cell cell) {
   _cells.add(cell);
 }

}
